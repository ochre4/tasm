Команда Описание Синтаксис ЗНАЧЕНИЕ_1 ЗНАЧЕНИЕ_2
mov Перемещает данные из одного регистра (ОПЕРАНД_2) в другой (ОПЕРАНД_1). Результат сохраняется в первый регистр (ОПЕРАНД_1). mov ЧИСЛО_1, ЧИСЛО_2 - Регистр;
- Переменная;		 - Регистр;
- Переменная;
- Число;
add Добавляет значение второго операнда (ОПЕРАНД_2) к первому операнду (ОПЕРАНД_1). Сохраняет значение в первый операнд (ОПЕРАНД_1). add ЧИСЛО_1, ЧИСЛО_2 - Регистр; - Регистр;
- Переменная;
- Число;
sub Вычитает значение второго операнда (ОПЕРАНД_2) из первого операнда (ОПЕРАНД_1). Сохраняет значение в первый операнд (ОПЕРАНД_1). sub ЧИСЛО_1, ЧИСЛО_2 - Регистр; - Регистр;
- Переменная;
- Число;
dec Уменьшает значение операнда (ОПЕРАНД_1) на 1 и сохраняет в этот же операнд (ОПЕРАНД_1). dec ЧИСЛО_1 - Регистр;
inc Увеличивает значение операнда (ОПЕРАНД_1) на 1 и сохраняет в этот же операнд (ОПЕРАНД_1). inc ЧИСЛО_1 - Регистр;
div Делит регистр "al" на значение операнда (ОПЕРАНД_1). Сохраняет результат в регистр "al". div ОПЕРАНД_1 - Регистр;
В обновлении будет добавлена возможность деления на число и переменную;
mul Умножает регистр "al" на значение операнда (ОПЕРАНД_1). Сохраняет результат в регистр "al". mul ОПЕРАНД_1 - Регистр;
В обновлении будет добавлена возможность деления на число и переменную;
cmp Сравнивает два операнда. cmp ОПЕРАНД_1, ОПЕРАНД_2 - Регистр; - Регистр;
- Переменная;
- Число;
je Переходит к метке (МЕТКА_1), если предыдущее сравнение (cmp) показало равенство. je МЕТКА_1 - Метка
jne Переходит к метке (МЕТКА_1), если предыдущее сравнение (cmp) показало неравенство. jne МЕТКА_1 - Метка;
jg Переходит к метке (МЕТКА_1), если предыдущее сравнение (cmp) показало, что ОПЕРАНД_1 больше ОПЕРАНД_2. jg МЕТКА_1 - Метка
jge Переходит к метке (МЕТКА_1), если предыдущее сравнение (cmp) показало, что ОПЕРАНД_1 больше или равен ОПЕРАНД_2. jge МЕТКА_1 - Метка;
jl Переходит к метке (МЕТКА_1), если предыдущее сравнение (cmp) показало, что ОПЕРАНД_1 меньше ОПЕРАНД_2. jl МЕТКА_1 - Метка
jle Переходит к метке (МЕТКА_1), если предыдущее сравнение (cmp) показало, что ОПЕРАНД_1 меньше или равен ОПЕРАНД_2. jle МЕТКА_1 - Метка;
ja Переходит к метке (МЕТКА_1), если предыдущее сравнение (cmp) показало, что беззнаковое значение ОПЕРАНД_1 больше беззнакового значения ОПЕРАНД_2. ja МЕТКА_1 - Метка
jae Переходит к метке (МЕТКА_1), если предыдущее сравнение (cmp) показало, что беззнаковое значение ОПЕРАНД_1 больше или равен беззнакового значения ОПЕРАНД_2. jae МЕТКА_1 - Метка;
jb Переходит к метке (МЕТКА_1), если предыдущее сравнение (cmp) показало, что беззнаковое значение ОПЕРАНД_1 меньше беззнакового значения ОПЕРАНД_2. jb МЕТКА_1 - Метка
jbe Переходит к метке (МЕТКА_1), если предыдущее сравнение (cmp) показало, что беззнаковое значение ОПЕРАНД_1 меньше или равен беззнакового значения ОПЕРАНД_2. jbe МЕТКА_1 - Метка;
loop Переход к метке (МЕТКА_1) и выполняет метку количество раз, которое записано в регистре "cl". loop МЕТКА_1 - Метка
